1. New Project > .netCore > ASP.net Core Web Application > API
2. Add Swashbuckle.AspNetCore plugin and add following code in ConfigureServices method in startup file


// Register the Swagger generator, defining 1 or more Swagger documents
	services.AddSwaggerGen(c =>
	{
		c.SwaggerDoc("v1", new Info { Title = "My API", Version = "v1" });
	});

	And following code in Configure method in startup file


	// Enable middleware to serve generated Swagger as a JSON endpoint.
	app.UseSwagger();

	// Enable middleware to serve swagger-ui (HTML, JS, CSS, etc.), 
	// specifying the Swagger JSON endpoint.
	app.UseSwaggerUI(c =>
	{
		c.SwaggerEndpoint("/swagger/v1/swagger.json", "My API V1");
	});
